Q: You are given a 3-digit number n, Find whether it is an Armstrong number or not.


int isArmstrong(int prgin) {
		int prg_state = prgin;
		int summition = 0; // Initialize summition
		while (prgin > 0) {
				int diff = prgin % 10;
				summition += diff * diff * diff;
				prgin /= 10; // Use /= for division assignment
		}
		if (summition == prg_state) {
				return 1;
		} else {
				return 0;
		}
}

Q: Given a positive integer, n. Find the factorial of n.


int factorial(int prgin) {
		if (prgin < 0) return -1; // Handle negative numbers
		int result = 1; // Initialize result
		for (int foo = 2; foo <= prgin; foo++) { // Start loop from 2
				result *= foo;
		}
		return result;
}

Q: Given two integer values n and r, the task is to find the value of Binomial Coefficient nCr.


int nCr(int prg_n, int prg_r) {
		// Check for invalid input
		if (prg_r < 0 || prg_r > prg_n) return 0;
		
		long long n_factorial = factorial(prg_n);
		long long r_factorial = factorial(prg_r);
		long long n_minus_r_factorial = factorial(prg_n - prg_r);
		
		return (int)(n_factorial / (r_factorial * n_minus_r_factorial));
}

Q: Given two integers a and b, You have to compute their LCM.


int findLCM(int foo, int bar) {
		int max_val = (foo > bar) ? foo : bar;
		int limit = foo * bar;

		for (int i = max_val; i <= limit; i++) {
				if (i % foo == 0 && i % bar == 0) {
						return i;
				}
		}
		return limit;
}

Q: Given two positive integers a and b, find GCD of a and b.


int findHCF(int foo, int bar) {
		while (bar != 0) {
				int temp = bar;
				bar = foo % bar;
				foo = temp;
		}
		return foo;
}

Q: Palindrome Array. Given an array arr, the task is to find whether the arr is palindrome or not. If the arr is palindrome then return true else return false.


class Solution {
  public:
    bool isPerfect(vector &arr) {
        int left_bit = 0;
        int right_bit = arr.size() - 1;
        while(left_bit <= right_bit){
            if(arr[left_bit] != arr[right_bit]){
                return (false);
            }
            left_bit++;
            right_bit--;
        }
        return (true);
    }
};

Q: Reverse an Array. You are given an array of integers arr[]. You have to reverse the given array.


class Solution {
  public:
    void reverseArray(vector &arr) {
        int left_bit = 0;
        int right_bit = arr.size() - 1;
        while(left_bit < right_bit){
            arr[left_bit] = arr[left_bit] ^ arr[right_bit];
            arr[right_bit] = arr[left_bit] ^ arr[right_bit];
            arr[left_bit] = arr[left_bit] ^ arr[right_bit];
            
            left_bit++;
            right_bit--;
        }
    }
};
