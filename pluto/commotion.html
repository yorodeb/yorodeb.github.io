int isArmstrong(int prgin) {
		int prg_state = prgin;
		int summition = 0; // Initialize summition
		while (prgin > 0) {
				int diff = prgin % 10;
				summition += diff * diff * diff;
				prgin /= 10; // Use /= for division assignment
		}
		if (summition == prg_state) {
				return 1;
		} else {
				return 0;
		}
}

int nCr(int prg_n, int prg_r) {
		// Check for invalid input
		if (prg_r < 0 || prg_r > prg_n) return 0;
		
		long long n_factorial = factorial(prg_n);
		long long r_factorial = factorial(prg_r);

int findLCM(int foo, int bar) {
		int max_val = (foo > bar) ? foo : bar;
		int limit = foo * bar;

		for (int i = max_val; i <= limit; i++) {

				}
		}

int findHCF(int foo, int bar) {

Q: Reverse an Array. You are given an array of integers arr[]. You have to reverse the given array.


class Solution {
  public:
    void reverseArray(vector &arr) {
        int left_bit = 0;
        int right_bit = arr.size() - 1;
        while(left_bit < right_bit){
            arr[left_bit] = arr[left_bit] ^ arr[right_bit];
            arr[right_bit] = arr[left_bit] ^ arr[right_bit];
            arr[left_bit] = arr[left_bit] ^ arr[right_bit];
            
            left_bit++;
            right_bit--;
        }
    }
};

Q: Reverse sub array. Given an array arr, you need to reverse a subarray of that array. The range of this subarray is given by indices l and r (1-based indexing).


class Solution {
  public:
    vector reverseSubArray(vector &arr, int l, int r) {
        l = l - 1;
        r = r - 1;
        while(l < r){
            arr[l] = arr[l] ^ arr[r];
            arr[r] = arr[l] ^ arr[r];
            arr[l] = arr[l] ^ arr[r];
            l++; r--;
        }
        return (arr);
    }
};

Q: Binary Search. Given an array of integers nums which is sorted in ascending order, and an integer target, write a function to search target in nums. If target exists, then return its index. Otherwise, return -1.


class Solution {
public:
    int search(vector& nums, int target) {
        int left_bit = 0; int right_bit = nums.size() - 1;
        while(left_bit <= right_bit){
            int mid_bit = (left_bit + right_bit) / 2;
            if(target == nums[mid_bit]){
                return (mid_bit);
            } else if(target < nums[mid_bit]){
                right_bit = mid_bit - 1;
            } else{
                left_bit = mid_bit + 1;
            }
        }
        return (-1);
    }
};

Q: Bubble Sort. Given an array, arr[]. Sort the array using bubble sort algorithm.


class Solution {
  public:
    void bubbleSort(vector& arr) {
        int size = arr.size();
        for(int i = 0; i < size - 1; i++){
            for(int j = 0; j < size - 1 - i; j++){
                if(arr[j] > arr[j+1]){
                    arr[j] = arr[j] ^ arr[j+1];
                    arr[j+1] = arr[j] ^ arr[j+1];
                    arr[j] = arr[j] ^ arr[j+1];
                }
            }
        }
    }
};

Q: Selection Sort. Given an array arr, use selection sort to sort arr[] in increasing order.


class Solution {
  public:
    void selectionSort(vector &arr) {
        int size = arr.size();
        for(int i = 0; i < size; i++){
            int MIN_IDX = i;
            for(int j = i + 1; j < size; j++){
                if(arr[j] < arr[MIN_IDX]){
                    MIN_IDX = j;
                }
            }
            swap(arr[i], arr[MIN_IDX]);
        }
    }
};

Q: Insertion Sort. Given an array arr[] of positive integers.The task is to complete the insertsort() function which is used to implement Insertion Sort.


class Solution {
  public:
    void insertionSort(vector& arr) {
        for(int i = 0; i < arr.size(); i++){
            int pivot = arr[i];
            int j = i - 1;
            
            while(j >= 0 && arr[j] > pivot){
                arr[j+1] = arr[j];
                j--;
            }
            arr[j+1] = pivot;
        }
    }
};
